// Creating tables
// You can define the tables with full schema names
Table cartridges.cartridge {
  id int [pk]
  name varchar
  svg varchar
  version varchar
  electrodes_count int
}


Table cartridges.electrode {
  id int [pk]
  cartridge_id int [ ref: > cartridges.cartridge.id ]
  svg_denomination varchar
 }
 
Table cartridges.feedback_type {
   id int [pk]
   name varchar
   data_type varchar
 }
 
Table cartridges.feedback {
   id int [pk]
   feedback_type_id int [ ref: > cartridges.feedback_type.id ]
   electrode_cartridge_id int [ ref: > cartridges.electrode.id ]
 }
 
Table protocols.protocol {
  id int [pk]
  name varchar
  frame_count int
  total_duration int
  mask_frame_id int [ ref: > protocols.frame.id ]
  date_created date
  version varchar
}

Table protocols.frame {
  id int [pk]
  protocol_id int [ ref: > protocols.protocol.id]
  duration int
  rank int
}

Table protocols.nested_protocols {
  id int [pk]
  parent_protocol_id int [ ref: > protocols.protocol.id]
  child_protocol_id int [ ref: > protocols.protocol.id]
}

Table protocols.frame_electrode {
  id int [pk]
  frame_id int [ ref: > protocols.frame.id ]
  value int
  electrode_id int [ ref: > cartridges.electrode.id]
}

Table protocols.frame_feature {
  id int [pk]
  frame_id int [ ref: > protocols.frame.id]
  electrode_id int [ ref: > cartridges.electrode.id]
  value int
}

Table protocols.frame_log {
  id int [pk]
  frame_id int [ ref: > protocols.frame.id]
  electrode_id int [ ref: > cartridges.electrode.id]
  value int
}

Table users.user {
  id int [pk]
  login varchar
  password varchar
}

Table protocols.authors {
  id int [pk]
  protocol_id int [ ref: > protocols.protocol.id]
  user_id varchar [ ref: > users.user.id]
  rank int
}

Table protocols.point_of_interest {
  id int [pk]
  frame_id int [ ref: > protocols.frame.id]
  electrode_id int [ ref: > cartridges.electrode.id]
  type_id varchar [ ref: > protocols.poi_types.id]
  protocol_id int [ ref: > protocols.protocol.id]
}

Table protocols.poi_types {
  id int [pk]
  name varchar
}
